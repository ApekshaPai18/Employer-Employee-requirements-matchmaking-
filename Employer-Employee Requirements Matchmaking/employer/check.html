const stars = document.querySelectorAll('.star'); const ratingValue = document.getElementById('rating-value'); const selectedRating = document.getElementById('selected-rating'); stars.forEach(star => { star.addEventListener('click', () => { const value = star.dataset.value; ratingValue.textContent = value; selectedRating.style.display = 'block'; // Mark all stars up to the clicked one as selected stars.forEach(innerStar => { innerStar.classList.remove('active'); if (innerStar.dataset.value <= value) { innerStar.classList.add('active'); } }); }); star.addEventListener('mouseover', () => { stars.forEach(innerStar => { if (!innerStar.classList.contains('active') && innerStar.dataset.value <= star.dataset.value) { innerStar.classList.add('active'); } }); }); star.addEventListener('mouseout', () => { if (star.classList.contains('active')) return; stars.forEach(innerStar => { innerStar.classList.remove('active'); }); }); });
